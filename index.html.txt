    <script type="module">
        import * as THREE from 'three';

        // --- Telegram WebApp ---
        let tg = window.Telegram?.WebApp || { initDataUnsafe: { user: { id: null } }, close: () => {}, expand: () => {}, themeParams: {} };
        tg.expand();
        
        const userId = tg.initDataUnsafe?.user?.id;
        document.getElementById('username').textContent = tg.initDataUnsafe?.user?.first_name || 'Стиляга';
        
        // --- 3D сцена ---
        const container = document.getElementById('canvas-container');
        const scene = new THREE.Scene();
        scene.background = null;
        
        const camera = new THREE.PerspectiveCamera(45, container.clientWidth / container.clientHeight, 0.1, 1000);
        camera.position.set(2, 1.5, 3.5);
        
        const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
        renderer.setSize(container.clientWidth, container.clientHeight);
        renderer.setClearColor(0x000000, 0);
        container.appendChild(renderer.domElement);
        
        // Освещение
        const ambientLight = new THREE.AmbientLight(0x404060);
        scene.add(ambientLight);
        const mainLight = new THREE.DirectionalLight(0xffeedd, 1);
        mainLight.position.set(3, 5, 2);
        scene.add(mainLight);
        
        // Стикмен
        const stickman = new THREE.Group();
        
        const bodyGeo = new THREE.CylinderGeometry(0.32, 0.32, 0.85, 24);
        const bodyMat = new THREE.MeshStandardMaterial({ color: 0x5a9eff });
        const body = new THREE.Mesh(bodyGeo, bodyMat);
        body.position.y = 0.45;
        stickman.add(body);
        
        const headGeo = new THREE.SphereGeometry(0.38, 48, 48);
        const headMat = new THREE.MeshStandardMaterial({ color: 0xffccaa });
        const head = new THREE.Mesh(headGeo, headMat);
        head.position.y = 0.95;
        stickman.add(head);
        
        const eyeMat = new THREE.MeshStandardMaterial({ color: 0xffffff });
        const leftEye = new THREE.Mesh(new THREE.SphereGeometry(0.09, 32, 32), eyeMat);
        leftEye.position.set(-0.13, 1.08, 0.36);
        const rightEye = new THREE.Mesh(new THREE.SphereGeometry(0.09, 32, 32), eyeMat);
        rightEye.position.set(0.13, 1.08, 0.36);
        stickman.add(leftEye, rightEye);
        
        const pupilMat = new THREE.MeshStandardMaterial({ color: 0x000000 });
        const leftPupil = new THREE.Mesh(new THREE.SphereGeometry(0.05, 32, 32), pupilMat);
        leftPupil.position.set(-0.13, 1.06, 0.44);
        const rightPupil = new THREE.Mesh(new THREE.SphereGeometry(0.05, 32, 32), pupilMat);
        rightPupil.position.set(0.13, 1.06, 0.44);
        stickman.add(leftPupil, rightPupil);
        
        const crownGeo = new THREE.CylinderGeometry(0.35, 0.45, 0.2, 8);
        const crownMat = new THREE.MeshStandardMaterial({ color: 0xffaa44, metalness: 0.8 });
        const crown = new THREE.Mesh(crownGeo, crownMat);
        crown.position.y = 1.22;
        stickman.add(crown);
        
        scene.add(stickman);
        
        // Подставка
        const platformGeo = new THREE.CylinderGeometry(0.9, 0.9, 0.08, 32);
        const platformMat = new THREE.MeshStandardMaterial({ color: 0xaa88ff, metalness: 0.5 });
        const platform = new THREE.Mesh(platformGeo, platformMat);
        platform.position.y = -0.25;
        scene.add(platform);
        
        // Слежение за курсором
        let mouseX = 0, mouseY = 0;
        window.addEventListener('mousemove', (event) => {
            const rect = container.getBoundingClientRect();
            const x = (event.clientX - rect.left) / rect.width;
            const y = (event.clientY - rect.top) / rect.height;
            mouseX = (x * 2 - 1) * 0.8;
            mouseY = (y * 2 - 1) * 0.4;
        });
        
        function animate() {
            requestAnimationFrame(animate);
            head.rotation.y += (mouseX - head.rotation.y) * 0.1;
            head.rotation.x += (-mouseY - head.rotation.x) * 0.08;
            renderer.render(scene, camera);
        }
        animate();
        
        window.addEventListener('resize', () => {
            const w = container.clientWidth, h = container.clientHeight;
            camera.aspect = w / h;
            camera.updateProjectionMatrix();
            renderer.setSize(w, h);
        });
        
        // Кнопка выбора подарка
        document.getElementById('select-gift-btn').onclick = () => {
            tg.sendData(JSON.stringify({ action: 'select_gift', user_id: userId }));
        };
        
        // Получение данных от бота
        tg.onEvent('message', (event) => {
            const data = JSON.parse(event.data);
            if (data.type === 'gifts') {
                // обновить список подарков
            }
        });
